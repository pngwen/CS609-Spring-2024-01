< Start >      ::= < Expression >

< Expression > ::= < Expression > "+" < Expression >
                   | < Expression > "-" < Expression >
                   | < Expression > "*" < Expression >
                   | < Expression > "/" < Expression >
                   | < Number >
                   
< Number >     ::= < Number > < Digit >
                   | < Digit >
               
< Digit >      ::= "0" | "1" | "2" | "3" | ... | "9"
